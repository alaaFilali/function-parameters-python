الحل: 

  
def prime_checker(number):
    is_prime = True
    for i in range(2, number):
        if number % i == 0:
            is_prime = False
    if is_prime:
        print("It's a prime number.")
    else:
        print("It's not a prime number")

n = int(input("Check this number: "))
prime_checker(number=n)

ببساطة شرح الحل كالتالي نحن نعلم ان العدد ان كان لا يقبل القسمة الا على نفسه و الرقم 1 فهو عدد اولي اذن نقوم بالتحقق ان كان ذلك العدد يقبل القسمة على اي عدد بين 1 و العدد نفسه فان كان كذلك فهو ليس عدد اولي والمتغير is_prime تتغير قيمته من True الى False وبناءا عليه نعرف اي جملة نطبع للمستخدم
